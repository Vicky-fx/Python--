import re
import requests
import pandas as pd
import os

headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 \
           (KHTML, like Gecko) Chrome/85.0.4183.83 Safari/537.36'}

news_list=pd.DataFrame(columns=['News_Title','Date','Source','Website'])

# page 为需要爬取的第N页（每页10条资讯）
def baidu(page):
    num=(page-1)*10
    # 百度资讯关键词（造纸+纸浆+木浆+行业）按时间排序
    url = 'https://www.baidu.com/s?tn=news&rtt=4&bsst=1&cl=2&wd=\
          %E9%80%A0%E7%BA%B8+%E7%BA%B8%E6%B5%86+%E6%9C%A8%E6%B5%86+%E8%A1%8C%E4%B8%9A&medium=0\
          &x_bfe_rqs=03E80000001&x_bfe_tjscore=0.100000&tngroupname=organic_news&newVideo=12&rsv_dl\
          =news_b_pn&pn='+str(num)
    res = requests.get(url, headers=headers).text
    
    p_href='<h3 class="news-title_1YtI1"><a href="(.*?)"'
    p_title='<h3 class="news-title_1YtI1">.*?<!--s-text-->(.*?)<!--/s-text-->'
    p_date='<h3 class="news-title_1YtI1">.*?<span class="c-color-gray2 c-font-normal">(.*?)</span>'
    p_source='<h3 class="news-title_1YtI1">.*?<span class="c-color-gray c-font-normal c-gap-right">(.*?)</span>'

    href=re.findall(p_href,res,re.S)
    title=re.findall(p_title,res,re.S)
    # 删除标题中<em>字符
    for i in range(len(title)):
        title[i]=title[i].strip()
        title[i]=re.sub('<.*?>','',title[i])
    date=re.findall(p_date,res,re.S)
    # 将中文日期标准化成 时间数据
    date=pd.to_datetime(date,format='%Y年%m月%d日 %H:%M')
    source=re.findall(p_source,res,re.S)
    news_list_i=pd.DataFrame({'News_Title':title,'Date':date,'Source':source,'Website':href})
    # print(news_list_i)
    return(news_list_i)

# 爬取前 10 页资讯标题
for i in range(10):
    baidu(i+1)
    # print('第'+str(i+1)+'页爬取成功')
    news_list_i=baidu(i+1)
    news_list=news_list.append(news_list_i,ignore_index=True)
